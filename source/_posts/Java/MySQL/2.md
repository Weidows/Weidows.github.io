# DQL-数据查询

> SELECT 列名 [AS 显示列名] FROM 表名 [AS 显示表名];

```SQL
-- 查询所有学生信息
SELECT
  *
FROM
  student;

-- 查询指定列
SELECT
  studentid,
  studentname
FROM
  student;

-- 这里是为列取别名(as关键词可以省略)
SELECT
  studentid AS `学 号`,
  studentname `姓 名`
FROM
  student;

-- as也可以为表取别名(并不会显示出来)
SELECT
  studentid AS `学 号`,
  studentname AS `姓 名`
FROM
  student AS `表`;

-- 通过CONCAT()函数拼接字符串
SELECT
  CONCAT('姓名: ', studentname) AS `拼接后的名字`
FROM
  student;
```
