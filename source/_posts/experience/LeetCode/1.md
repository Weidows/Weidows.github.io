      def longestPalindromeSubseq(self, s):
          """
          :type s: str
          :rtype: int
          """
          if len(s) < 2:
              return len(s)

          n = len(s)
          dp = [[0] * n for _ in range(n)]
          for i in range(n - 1, -1, -1):
              dp[i][i] = 1
              for j in range(i + 1, n):
